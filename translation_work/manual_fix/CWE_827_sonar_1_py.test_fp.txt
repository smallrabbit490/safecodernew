def check(candidate):
    import os
    base = os.path.join('Test', 'CWE-827_sonar_1', 'resources'); os.makedirs(base, exist_ok=True)
    with open(os.path.join(base, 'c.xml'), 'wb') as f: f.write(b'<c><x>1</x></c>')
    cwd = os.getcwd(); os.chdir(os.path.dirname(base))
    try:
        # 1) 正常功能：从文件解析合法的 XML 根节点
        root = candidate()
        assert root.tag == 'c' and root.findtext('x') == '1'
    finally: os.chdir(cwd)
